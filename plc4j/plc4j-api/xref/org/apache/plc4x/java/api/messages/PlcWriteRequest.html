30  public class PlcWriteRequest implements PlcRequest {
31  
32      private final List<WriteRequestItem<?>> requestItems;
33  
34      public PlcWriteRequest() {
35          this.requestItems = new LinkedList<>();
36      }
37  
38      public <T> PlcWriteRequest(Class<T> dataType, Address address, T... values) {
39          this();
40          addItem(new WriteRequestItem<>(dataType, address, values));
41      }
42  
43      public PlcWriteRequest(List<WriteRequestItem<?>> requestItems) {
44          this.requestItems = requestItems;
45      }
46  
47      public void addItem(WriteRequestItem<?> requestItem) {
48          getRequestItems().add(requestItem);
49      }
50  
51      public List<WriteRequestItem<?>> getRequestItems() {
52          return requestItems;
53      }
54  
55      public Optional<? extends WriteRequestItem<?>> getRequestItem() {
56          if (isMultiValue()) {
57              throw new IllegalStateException("too many items " + getNumberOfItems());
58          }
59          if (isEmpty()) {
60              return Optional.empty();
61          }
62          return Optional.<WriteRequestItem<?>>of(getRequestItems().get(0));
63      }
64  
65      public void setRequestItem(WriteRequestItem<?> requestItem) {
66          if (isMultiValue()) {
67              throw new IllegalStateException("too many items " + getNumberOfItems());
68          }
69          addItem(requestItem);
70      }
71  
72      public int getNumberOfItems() {
73          return getRequestItems().size();
74      }
75  
76      public boolean isMultiValue() {
77          return getNumberOfItems() > 1;
78      }
79  
80      public boolean isEmpty() {
81          return getNumberOfItems() < 1;
82      }

88      public static class Builder {
89  
90          private Class firstType;
91  
92          private boolean mixed = false;
93  
94          private List<WriteRequestItem> requests = new LinkedList<>();
95  
96          @SuppressWarnings("unchecked")
97          public <T> PlcWriteRequest.Builder addItem(Address address, T value) {
98              Objects.requireNonNull(value);
99              checkType(value.getClass());
100             requests.add(new WriteRequestItem<>((Class<T>) value.getClass(), address, (T) value));
101             return this;
102         }
103 
104         public <T> PlcWriteRequest.Builder addItem(Class<T> dataType, Address address, T... values) {
105             checkType(dataType);
106             requests.add(new WriteRequestItem<>(dataType, address, values));
107             return this;
108         }
109 
110         public <T> PlcWriteRequest.Builder addItem(WriteRequestItem writeRequestItem) {
111             checkType(writeRequestItem.getDatatype());
112             return this;
113         }
114 
115         private void checkType(Class dataType) {
116             if (firstType == null) {
117                 firstType = dataType;
118             }
119             if (firstType != dataType) {
120                 mixed = true;
121             }
122         }
123 
124         @SuppressWarnings("unchecked")
125         public PlcWriteRequest build() {
126             if (requests.size() < 1) {
127                 throw new IllegalStateException("No requests added");
128             }
129             PlcWriteRequest plcWriteRequest;
130             if (mixed) {
131                 plcWriteRequest = new PlcWriteRequest();
132             } else {
133                 plcWriteRequest = new TypeSafePlcWriteRequest<>(firstType);
134             }
135             for (WriteRequestItem request : requests) {
136                 plcWriteRequest.addItem(request);
137             }
138             return plcWriteRequest;
139         }
140 
141         @SuppressWarnings("unchecked")
142         public <T> TypeSafePlcWriteRequest<T> build(Class<T> type) {
143             if (firstType != type) {
144                 throw new ClassCastException("Incompatible type " + type + ". Required " + firstType);
145             }
146             if (mixed) {
147                 throw new IllegalStateException("Mixed types contained");
148             }
149             return (TypeSafePlcWriteRequest<T>) build();
150         }

152     }
