19  package org.apache.plc4x.java.modbus.connection;

21  import org.apache.commons.lang3.StringUtils;
22  import org.apache.plc4x.java.api.connection.PlcReader;

24  import org.apache.plc4x.java.api.exceptions.PlcException;
25  import org.apache.plc4x.java.api.messages.*;
26  import org.apache.plc4x.java.api.model.Address;
27  import org.apache.plc4x.java.base.connection.AbstractPlcConnection;
28  import org.apache.plc4x.java.base.connection.ChannelFactory;
29  import org.apache.plc4x.java.modbus.model.ModbusAddress;
30  import org.slf4j.Logger;
31  import org.slf4j.LoggerFactory;
32  
33  import java.util.concurrent.CompletableFuture;
34  import java.util.regex.Matcher;
35  import java.util.regex.Pattern;
36  
37  public abstract class BaseModbusPlcConnection extends AbstractPlcConnection implements PlcReader, PlcWriter {
38  
39      private static final Pattern MODBUS_ADDRESS_PATTERN =
40          Pattern.compile("^(?<memoryArea>.*?)/(?<byteOffset>\\d{1,4})(?:/(?<bitOffset>\\d))?");
41  
42      private static final Logger logger = LoggerFactory.getLogger(BaseModbusPlcConnection.class);
43  
44      protected BaseModbusPlcConnection(ChannelFactory channelFactory, String params) {
45          super(channelFactory, true);
46  
47          if (!StringUtils.isEmpty(params)) {
48              for (String param : params.split("&")) {
49                  String[] paramElements = param.split("=");
50                  String paramName = paramElements[0];
51                  if (paramElements.length == 2) {
52                      String paramValue = paramElements[1];
53                      switch (paramName) {
54                          default:
55                              logger.debug("Unknown parameter {} with value {}", paramName, paramValue);
56                      }
57                  } else {
58                      logger.debug("Unknown no-value parameter {}", paramName);
59                  }
60              }
61          }
62      }
63  
64      @Override
65      public Address parseAddress(String addressString) throws PlcException {
66          Matcher addressMatcher = MODBUS_ADDRESS_PATTERN.matcher(addressString);
67          if (addressMatcher.matches()) {
68              /*int datablockNumber = Integer.parseInt(datablockAddressMatcher.group("blockNumber"));
69              int datablockByteOffset = Integer.parseInt(datablockAddressMatcher.group("byteOffset"));*/
70              return new ModbusAddress();
71          }
72          return null;
73      }
74  
75      @Override
76      public CompletableFuture<PlcReadResponse> read(PlcReadRequest readRequest) {
77          CompletableFuture<PlcReadResponse> readFuture = new CompletableFuture<>();
78          PlcRequestContainer<PlcReadRequest, PlcReadResponse> container =
79              new PlcRequestContainer<>(readRequest, readFuture);
80          channel.writeAndFlush(container);
81          return readFuture;
82      }
83  
84      @Override
85      public CompletableFuture<PlcWriteResponse> write(PlcWriteRequest writeRequest) {
86          CompletableFuture<PlcWriteResponse> writeFuture = new CompletableFuture<>();
87          PlcRequestContainer<PlcWriteRequest, PlcWriteResponse> container =
88              new PlcRequestContainer<>(writeRequest, writeFuture);
89          channel.writeAndFlush(container);
90          return writeFuture;
91      }
92  
93  }
