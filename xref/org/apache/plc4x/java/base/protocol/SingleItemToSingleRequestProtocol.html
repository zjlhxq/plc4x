43  /**
44   * This layer can be used to split a {@link org.apache.plc4x.java.api.messages.PlcRequest} which addresses multiple {@link PlcField}s into multiple subsequent {@link org.apache.plc4x.java.api.messages.PlcRequest}s.
45   */
46  public class SingleItemToSingleRequestProtocol extends ChannelDuplexHandler {
47  
48      public static final Logger LOGGER = LoggerFactory.getLogger(SingleItemToSingleRequestProtocol.class);
49  
50      private final Timer timer;
51  
52      // TODO: maybe better get from map
53      private long defaultReceiveTimeout;
54  
55      private PendingWriteQueue queue;
56  
57      private ConcurrentMap<PlcRequestContainer<InternalPlcRequest, InternalPlcResponse<?>>, Timeout> scheduledTimeouts;
58  
59      // Map to track send subcontainers
60      private ConcurrentMap<Integer, PlcRequestContainer<InternalPlcRequest, InternalPlcResponse<?>>> sentButUnacknowledgedSubContainer;
61  
62      // Map to map tdpu to original parent container
63      // TODO: currently this could be supplied via param, only reason to keep would be for statistics.
64      private ConcurrentMap<Integer, PlcRequestContainer<InternalPlcRequest, InternalPlcResponse<?>>> correlationToParentContainer;
65  
66      // Map to track tdpus per container
67      // TODO: currently this could be supplied via param, only reason to keep would be for statistics.
68      private ConcurrentMap<PlcRequestContainer<?, ?>, Set<Integer>> containerCorrelationIdMap;
69  
70      // Map to track a list of responses per parent container
71      private ConcurrentMap<PlcRequestContainer<?, ?>, Queue<InternalPlcResponse<?>>> responsesToBeDelivered;
72  
73      private AtomicInteger correlationIdGenerator;
74  
75      // TODO: maybe put in map per day or per hour
76      private AtomicLong deliveredContainers;
77  
78      private AtomicLong erroredContainers;
79  
80      private AtomicLong deliveredItems;
81  
82      private AtomicLong erroredItems;

84      public SingleItemToSingleRequestProtocol(Timer timer) {
85          this(timer, true);
86      }
87  
88      public SingleItemToSingleRequestProtocol(Timer timer, boolean betterImplementationPossible) {
89          this(timer, TimeUnit.SECONDS.toMillis(30), betterImplementationPossible);
90      }
91  
92      public SingleItemToSingleRequestProtocol(Timer timer, long defaultReceiveTimeout, boolean betterImplementationPossible) {
93          this.timer = timer;
94          this.defaultReceiveTimeout = defaultReceiveTimeout;
95          if (betterImplementationPossible) {
96              String callStack = Arrays.stream(Thread.currentThread().getStackTrace())
97                  .map(StackTraceElement::toString)
98                  .collect(Collectors.joining("\n"));
99              LOGGER.warn("Unoptimized Usage of {} detected at:\n{}", this.getClass(), callStack);
100             LOGGER.info("Consider implementing item splitting native to the protocol.");
101         }
102     }

104     @Override
105     public void channelRegistered(ChannelHandlerContext ctx) throws Exception {
106         this.queue = new PendingWriteQueue(ctx);
107         this.scheduledTimeouts = new ConcurrentHashMap<>();
108         this.sentButUnacknowledgedSubContainer = new ConcurrentHashMap<>();
109         this.correlationToParentContainer = new ConcurrentHashMap<>();
110         this.containerCorrelationIdMap = new ConcurrentHashMap<>();
111         this.responsesToBeDelivered = new ConcurrentHashMap<>();
112         this.correlationIdGenerator = new AtomicInteger();
113         this.deliveredItems = new AtomicLong();
114         this.erroredItems = new AtomicLong();
115         this.deliveredContainers = new AtomicLong();
116         this.erroredContainers = new AtomicLong();
117         super.channelRegistered(ctx);
118     }
119 
120     @Override
121     public void channelUnregistered(ChannelHandlerContext ctx) throws Exception {
122         this.queue.removeAndWriteAll();
123         this.scheduledTimeouts.clear();
124         this.sentButUnacknowledgedSubContainer.clear();
125         this.correlationToParentContainer.clear();
126         this.containerCorrelationIdMap.clear();
127         this.responsesToBeDelivered.clear();
128         this.correlationIdGenerator.set(0);
129         this.deliveredItems.set(0);
130         this.erroredItems.set(0);
131         this.deliveredContainers.set(0);
132         this.erroredContainers.set(0);
133         super.channelUnregistered(ctx);
134     }

136     @Override
137     public void channelInactive(ChannelHandlerContext ctx) throws Exception {
138         // Send everything so we get a proper failure for those pending writes
139         this.queue.removeAndWriteAll();
140         this.timer.stop();
141         this.scheduledTimeouts.clear();
142         this.sentButUnacknowledgedSubContainer.clear();
143         this.correlationToParentContainer.clear();
144         this.containerCorrelationIdMap.clear();
145         this.responsesToBeDelivered.clear();
146         this.correlationIdGenerator.set(0);
147         this.deliveredItems.set(0);
148         this.erroredItems.set(0);
149         this.deliveredContainers.set(0);
150         this.erroredContainers.set(0);
151         super.channelInactive(ctx);
152     }

158     protected void tryFinish(Integer currentTdpu, InternalPlcResponse msg, CompletableFuture<InternalPlcResponse<?>> originalResponseFuture) {
159         deliveredItems.incrementAndGet();
160         PlcRequestContainer<InternalPlcRequest, InternalPlcResponse<?>> subPlcRequestContainer = sentButUnacknowledgedSubContainer.remove(currentTdpu);
161         LOGGER.info("{} got acknowledged", subPlcRequestContainer);
162         PlcRequestContainer<InternalPlcRequest, InternalPlcResponse<?>> originalPlcRequestContainer = correlationToParentContainer.remove(currentTdpu);
163         if (originalPlcRequestContainer == null) {
164             LOGGER.warn("Unrelated package received {}", msg);
165             return;
166         }
167         Queue<InternalPlcResponse<?>> correlatedResponseItems = responsesToBeDelivered.computeIfAbsent(originalPlcRequestContainer, ignore -> new ConcurrentLinkedQueue<>());
168         correlatedResponseItems.add(msg);
169         Set<Integer> integers = containerCorrelationIdMap.get(originalPlcRequestContainer);
170         integers.remove(currentTdpu);
171         if (integers.isEmpty()) {
172             deliveredContainers.incrementAndGet();
173             Timeout timeout = scheduledTimeouts.remove(originalPlcRequestContainer);
174             if (timeout != null) {
175                 timeout.cancel();
176             }
177 
178             InternalPlcResponse<?> plcResponse;
179             if (originalPlcRequestContainer.getRequest() instanceof InternalPlcReadRequest) {
180                 InternalPlcReadRequest internalPlcReadRequest = (InternalPlcReadRequest) originalPlcRequestContainer.getRequest();
181                 HashMap<String, Pair<PlcResponseCode, FieldItem>> fields = new HashMap<>();
182 
183                 correlatedResponseItems.stream()
184                     .map(InternalPlcReadResponse.class::cast)
185                     .map(InternalPlcReadResponse::getValues)
186                     .forEach(stringPairMap -> stringPairMap.forEach(fields::put));
187 
188                 plcResponse = new DefaultPlcReadResponse(internalPlcReadRequest, fields);
189             } else if (originalPlcRequestContainer.getRequest() instanceof InternalPlcWriteRequest) {
190                 InternalPlcWriteRequest internalPlcWriteRequest = (InternalPlcWriteRequest) originalPlcRequestContainer.getRequest();
191                 HashMap<String, PlcResponseCode> values = new HashMap<>();
192 
193                 correlatedResponseItems.stream()
194                     .map(InternalPlcWriteResponse.class::cast)
195                     .map(InternalPlcWriteResponse::getValues)
196                     .forEach(stringPairMap -> stringPairMap.forEach(values::put));
197 
198                 plcResponse = new DefaultPlcWriteResponse(internalPlcWriteRequest, values);
199             } else {
200                 errored(currentTdpu, new PlcProtocolException("Unknown type detected " + originalPlcRequestContainer.getRequest().getClass()), originalResponseFuture);
201                 return;
202             }
203             responsesToBeDelivered.remove(originalPlcRequestContainer);
204             containerCorrelationIdMap.remove(originalPlcRequestContainer);
205             originalResponseFuture.complete(plcResponse);
206         }
207     }
208 
209     protected void errored(Integer currentTdpu, Throwable throwable, CompletableFuture<InternalPlcResponse<?>> originalResponseFuture) {
210         erroredItems.incrementAndGet();
211         PlcRequestContainer<InternalPlcRequest, InternalPlcResponse<?>> subPlcRequestContainer = sentButUnacknowledgedSubContainer.remove(currentTdpu);
212         LOGGER.info("{} got errored", subPlcRequestContainer);
213 
214 
215         PlcRequestContainer<InternalPlcRequest, InternalPlcResponse<?>> originalPlcRequestContainer = correlationToParentContainer.remove(currentTdpu);
216         if (originalPlcRequestContainer == null) {
217             LOGGER.warn("Unrelated error received tdpu:{}", currentTdpu, throwable);
218         } else {
219             erroredContainers.incrementAndGet();
220             Timeout timeout = scheduledTimeouts.remove(originalPlcRequestContainer);
221             if (timeout != null) {
222                 timeout.cancel();
223             }
224             responsesToBeDelivered.remove(originalPlcRequestContainer);
225 
226             Set<Integer> tdpus = containerCorrelationIdMap.remove(originalPlcRequestContainer);
227             if (tdpus != null) {
228                 tdpus.forEach(tdpu -> {
229                     // TODO: technically the other items didn't error so do we increment?
230                     //erroredItems.incrementAndGet();
231                     sentButUnacknowledgedSubContainer.remove(tdpu);
232                     correlationToParentContainer.remove(tdpu);
233                 });
234             }
235 
236             LOGGER.warn("PlcRequestContainer {} and correlationId {} failed ", correlationToParentContainer, currentTdpu, throwable);
237             originalResponseFuture.completeExceptionally(throwable);
238         }
239     }

245     @Override
246     public void write(ChannelHandlerContext ctx, Object msg, ChannelPromise promise) throws Exception {
247         if (msg instanceof PlcRequestContainer) {
248             @SuppressWarnings("unchecked")
249             PlcRequestContainer<InternalPlcRequest, InternalPlcResponse<?>> in = (PlcRequestContainer<InternalPlcRequest, InternalPlcResponse<?>>) msg;
250             Set<Integer> tdpus = containerCorrelationIdMap.computeIfAbsent(in, plcRequestContainer -> ConcurrentHashMap.newKeySet());
251 
252             Timeout timeout = timer.newTimeout(timeout_ -> handleTimeout(timeout_, in, tdpus, System.nanoTime()), defaultReceiveTimeout, TimeUnit.MILLISECONDS);
253             scheduledTimeouts.put(in, timeout);
254 
255             // Create a promise that has to be called multiple times.
256             PromiseCombiner promiseCombiner = new PromiseCombiner();
257             InternalPlcRequest request = in.getRequest();
258             if (request instanceof InternalPlcFieldRequest) {
259                 InternalPlcFieldRequest internalPlcFieldRequest = (InternalPlcFieldRequest) request;
260 
261                 if (internalPlcFieldRequest instanceof InternalPlcReadRequest) {
262                     InternalPlcReadRequest internalPlcReadRequest = (InternalPlcReadRequest) internalPlcFieldRequest;
263                     internalPlcReadRequest.getNamedFields().forEach(field -> {
264                         ChannelPromise subPromise = new DefaultChannelPromise(promise.channel());
265 
266                         Integer tdpu = correlationIdGenerator.getAndIncrement();
267                         CompletableFuture<InternalPlcResponse> correlatedCompletableFuture = new CompletableFuture<>();
268                         // Important: don't chain to above as we want the above to be completed not the result of when complete
269                         correlatedCompletableFuture
270                             .thenApply(InternalPlcResponse.class::cast)
271                             .whenComplete((internalPlcResponse, throwable) -> {
272                                 if (throwable != null) {
273                                     errored(tdpu, throwable, in.getResponseFuture());
274                                 } else {
275                                     tryFinish(tdpu, internalPlcResponse, in.getResponseFuture());
276                                 }
277                             });
278                         PlcRequestContainer<CorrelatedPlcReadRequest, InternalPlcResponse> correlatedPlcRequestContainer = new PlcRequestContainer<>(CorrelatedPlcReadRequest.of(field, tdpu), correlatedCompletableFuture);
279                         correlationToParentContainer.put(tdpu, in);
280                         queue.add(correlatedPlcRequestContainer, subPromise);
281                         if (!tdpus.add(tdpu)) {
282                             throw new IllegalStateException("AtomicInteger should not create duplicated ids: " + tdpu);
283                         }
284                         promiseCombiner.add((Future) subPromise);
285                     });
286                 } else if (internalPlcFieldRequest instanceof InternalPlcWriteRequest) {
287                     InternalPlcWriteRequest internalPlcWriteRequest = (InternalPlcWriteRequest) internalPlcFieldRequest;
288                     internalPlcWriteRequest.getNamedFieldTriples().forEach(fieldItemTriple -> {
289                         ChannelPromise subPromise = new DefaultChannelPromise(promise.channel());
290 
291                         Integer tdpu = correlationIdGenerator.getAndIncrement();
292                         CompletableFuture<InternalPlcResponse> correlatedCompletableFuture = new CompletableFuture<>()
293                             .thenApply(InternalPlcResponse.class::cast)
294                             .whenComplete((internalPlcResponse, throwable) -> {
295                                 if (throwable != null) {
296                                     errored(tdpu, throwable, in.getResponseFuture());
297                                 } else {
298                                     tryFinish(tdpu, internalPlcResponse, in.getResponseFuture());
299                                 }
300                             });
301                         PlcRequestContainer<CorrelatedPlcWriteRequest, InternalPlcResponse> correlatedPlcRequestContainer = new PlcRequestContainer<>(CorrelatedPlcWriteRequest.of(fieldItemTriple, tdpu), correlatedCompletableFuture);
302                         correlationToParentContainer.put(tdpu, in);
303                         queue.add(correlatedPlcRequestContainer, subPromise);
304                         if (!tdpus.add(tdpu)) {
305                             throw new IllegalStateException("AtomicInteger should not create duplicated ids: " + tdpu);
306                         }
307                         promiseCombiner.add((Future) subPromise);
308                     });
309                     // TODO: add sub/unsub
310                 } else {
311                     throw new PlcProtocolException("Unmapped request type " + request.getClass());
312                 }
313             } else {
314                 ChannelPromise subPromise = new DefaultChannelPromise(promise.channel());
315                 queue.add(msg, subPromise);
316                 promiseCombiner.add((Future) subPromise);
317             }
318 
319             promiseCombiner.finish(promise);
320 
321             // Start sending the queue content.
322             trySendingMessages(ctx);
323         } else {
324             super.write(ctx, msg, promise);
325         }
326     }

332     @SuppressWarnings("unchecked")
333     protected synchronized void trySendingMessages(ChannelHandlerContext ctx) {
334         while (queue.size() > 0) {
335             // Get the RequestItem that is up next in the queue.
336             PlcRequestContainer currentItem = (PlcRequestContainer) queue.current();
337             InternalPlcRequest request = currentItem.getRequest();
338 
339             // Send the TPDU.
340             try {
341                 ChannelFuture channelFuture = queue.removeAndWrite();
342                 ctx.flush();
343                 if (channelFuture == null) {
344                     break;
345                 }
346             } catch (Exception e) {
347                 LOGGER.error("Error sending more queues messages", e);
348                 ctx.fireExceptionCaught(e);
349             }
350 
351             if (request instanceof CorrelatedPlcRequest) {
352                 CorrelatedPlcRequest correlatedPlcRequest = (CorrelatedPlcRequest) request;
353 
354                 // Add it to the list of sentButUnacknowledgedSubContainer.
355                 sentButUnacknowledgedSubContainer.put(correlatedPlcRequest.getTdpu(), currentItem);
356 
357                 LOGGER.debug("container with id {}sent: ", correlatedPlcRequest.getTdpu(), currentItem);
358             }
359         }
360         ctx.flush();
361     }

363     private void handleTimeout(Timeout timeout, PlcRequestContainer<InternalPlcRequest, InternalPlcResponse<?>> in, Set<Integer> tdpus, long scheduledAt) {
364         if (timeout.isCancelled()) {
365             LOGGER.debug("container {} with timeout {} got canceled", in, timeout);
366             return;
367         }
368         LOGGER.warn("container {} timed out:{}", in, timeout);
369         erroredContainers.incrementAndGet();
370         responsesToBeDelivered.remove(in);
371         containerCorrelationIdMap.remove(in);
372         tdpus.forEach(tdpu -> {
373             erroredItems.incrementAndGet();
374             sentButUnacknowledgedSubContainer.remove(tdpu);
375             correlationToParentContainer.remove(tdpu);
376         });
377         in.getResponseFuture().completeExceptionally(new PlcTimeoutException(System.nanoTime() - scheduledAt));
378     }

385     protected static class CorrelatedPlcReadRequest extends DefaultPlcReadRequest implements CorrelatedPlcRequest {
386 
387         protected final int tdpu;
388 
389         protected CorrelatedPlcReadRequest(LinkedHashMap<String, PlcField> fields, int tdpu) {
390             super(fields);
391             this.tdpu = tdpu;
392         }
393 
394         protected static CorrelatedPlcReadRequest of(Pair<String, PlcField> stringPlcFieldPair, int tdpu) {
395             LinkedHashMap<String, PlcField> fields = new LinkedHashMap<>();
396             fields.put(stringPlcFieldPair.getKey(), stringPlcFieldPair.getValue());
397             return new CorrelatedPlcReadRequest(fields, tdpu);
398         }
399 
400         @Override
401         public int getTdpu() {
402             return tdpu;
403         }
404     }

406     protected static class CorrelatedPlcWriteRequest extends DefaultPlcWriteRequest implements CorrelatedPlcRequest {
407 
408         private final int tdpu;
409 
410         public CorrelatedPlcWriteRequest(LinkedHashMap<String, Pair<PlcField, FieldItem>> fields, int tdpu) {
411             super(fields);
412             this.tdpu = tdpu;
413         }
414 
415         public static CorrelatedPlcWriteRequest of(Triple<String, PlcField, FieldItem> fieldItemTriple, int tdpu) {
416             LinkedHashMap<String, Pair<PlcField, FieldItem>> fields = new LinkedHashMap<>();
417             fields.put(fieldItemTriple.getLeft(), Pair.of(fieldItemTriple.getMiddle(), fieldItemTriple.getRight()));
418             return new CorrelatedPlcWriteRequest(fields, tdpu);
419         }
420 
421         @Override
422         public int getTdpu() {
423             return tdpu;
424         }
425     }

427     // TODO: maybe export to jmx
428     public Map<String, Number> getStatistics() {
429         HashMap<String, Number> statistics = new HashMap<>();
430         statistics.put("queue", queue.size());
431         statistics.put("sentButUnacknowledgedSubContainer", sentButUnacknowledgedSubContainer.size());
432         statistics.put("correlationToParentContainer", correlationToParentContainer.size());
433         statistics.put("containerCorrelationIdMap", containerCorrelationIdMap.size());
434         statistics.put("responsesToBeDelivered", responsesToBeDelivered.size());
435         statistics.put("correlationIdGenerator", correlationIdGenerator.get());
436         statistics.put("deliveredItems", deliveredItems.get());
437         statistics.put("erroredItems", erroredItems.get());
438         statistics.put("deliveredContainers", deliveredContainers.get());
439         statistics.put("erroredContainers", erroredContainers.get());
440         return statistics;
441     }
